 23def enrich_dicts_with_information(list_dict: List[Dict]):
 24    """based on available information of each spectra, additional information is calculated and stored.
 25
 26    :param list_dict: list of dictonaries containging spectra and meta data
 27    :type list_dict: List[Dict]
 28    :return: same dicts with additional information
 29    :rtype: List[Dict]
 30    """
 31    for da in list_dict:
 32        da["scores_norm_sorted"] = sorted(da["scores_norm"], reverse=True)
 33        da["scores_norm_mean"] = np.mean(da["scores_norm"])
 34        da["scores_norm_median"] = np.median(da["scores_norm"])
 35        da["explained_variance"] = explained_variance_score(
 36            da["spectrum"], da["denoised"]
 37        )
 38        da["cosine_similarity"] = cosine_similarity(da["spectrum"], da["denoised"])
 39        da["scores_norm_abs"] = np.sqrt(np.sum(np.square(da["scores_norm"])))
 40    return list_dict

 43def split_spectra_one_vs_all(
 44    data: List[Dict], isotopes: List[str], excluded_isotope: str
 45):
 46    """
 47    Split data by label into two datasets: one isotope and all others.
 48
 49    :param data: spectral data
 50    :type data: List[Dict]
 51    :param isotopes: list of all isotopes found in data as labels
 52    :type isotopes: List[str]
 53    :param excluded_isotope: _description_
 54    :type excluded_isotope: str
 55
 56    :return: data_one: data of excluded_isotope
 57            isotope_one: name of excluded_isotope (as array)
 58            data_other: data from all isotopes but excluded_isotope
 59            isotopes_other: names of all isotopes but excluded_isotopes (as array)
 60    :rtype: Tuple
 61    """
 62
 63    if excluded_isotope not in isotopes:
 64        raise ValueError(
 65            f"You are trying to remove {excluded_isotope} from the data but \
 66                         it does not exist! Given isotopes: {isotopes}"
 67        )
 68
 69    # divide isotopes into two arrays: 1. excluded_isotope, 2. all others
 70    isotope_one = np.array([excluded_isotope])
 71    isotopes_other = np.array([iso for iso in isotopes if iso != excluded_isotope])
 72    print(f"\n Excluded isotope: {isotope_one}")
 73
 74    data_one = [
 75        x for x in data if excluded_isotope in x["labels"]
 76    ]  # all data of excluded_isotope
 77    data_other = [
 78        x for x in data if excluded_isotope not in x["labels"]
 79    ]  # all data of other isotopes
 80
 81    return data_one, isotope_one, data_other, isotopes_other

 84def simulate_outlier(isotope_outlier: str):
 85    """
 86    Trains a dimensionality reduction model with all outliers except for isotope_outlier.
 87    Then, the model is applied to all known data, but also to unknown data (of isotope_outlier).
 88    We record the information how unknown spectra appears in latent space
 89    and construct two arrays with information on the known and unknown isotopes.
 90
 91    :param isotope_outlier: name of isotope that should be excluded from training
 92    :type isotope_outlier: str
 93    :return: np.array(), np.array()
 94    """
 95
 96    data, all_isotopes = load_spectral_data(dir_numpy_ready, GlobalVariables.dets_tr)
 97    all_isotopes = [
 98        x[0] for x in all_isotopes
 99    ]  # only single-label spectra allowed for model training
100
101    data = remove_empty_or_negative_spectra(data)
102
103    # Option to exclude one isotope (for Outlier analysis)
104    excluded_isotope = isotope_outlier
105    data_unknown, isotope_unknown, data_known, isotopes_known = (
106        split_spectra_one_vs_all(data, all_isotopes, excluded_isotope)
107    )
108    GlobalVariables.isotopes_tr = (
109        isotopes_known  # save only known isotopes to global variable for training
110    )
111
112    # Training for dimensionality reduction
113    train_dim_model(data_known, save_results=False)
114
115    data_known = apply_dim_model(data_known)
116    data_unknown = apply_dim_model(data_unknown)
117
118    # Feature Engineering
119    for da in [data_known, data_unknown]:
120        da = enrich_dicts_with_information(da)
121
122    x_known = []
123    for da in data_known:
124        combined_list = da["scores_norm_sorted"] + [
125            da["scores_norm_mean"],
126            da["scores_norm_median"],
127            da["explained_variance"],
128            da["cosine_similarity"],
129            da["scores_norm_abs"],
130        ]
131        x_known.append(combined_list)
132
133    x_unknown = []
134    for da in data_unknown:
135        combined_list = da["scores_norm_sorted"] + [
136            da["scores_norm_mean"],
137            da["scores_norm_median"],
138            da["explained_variance"],
139            da["cosine_similarity"],
140            da["scores_norm_abs"],
141        ]
142        x_unknown.append(combined_list)
143
144    return x_known, x_unknown

147def check_column_names_match(x_combined, labels_combined, column_names):
148    """
149    Validates that the number of rows in `x_combined` and `labels_combined` match and that the combined
150    array has the same number of columns as specified in `column_names`.
151
152    Parameters
153    ----------
154    x_combined : np.ndarray
155        The first array to be combined, typically containing feature data.
156    labels_combined : np.ndarray
157        The second array to be combined, typically containing labels or additional data columns.
158    column_names : list of str
159        A list of column names that should match the total number of columns in the horizontally stacked array.
160
161    Raises
162    ------
163    ValueError
164        If `x_combined` and `labels_combined` do not have the same number of rows.
165    ValueError
166        If the combined array has a different number of columns than the length of `column_names`.
167
168    Notes
169    -----
170    This function horizontally stacks `x_combined` and `labels_combined`, then verifies the column count
171    against `column_names`. If the count does not match, it raises an error to prompt manual adjustment.
172    """
173    # Check if number of rows match
174    if x_combined.shape[0] != labels_combined.shape[0]:
175        raise ValueError(
176            "x_combined and labels_combined must have the same number of rows"
177        )
178
179    # Stack the arrays horizontally
180    combined = np.hstack((x_combined, labels_combined))
181
182    # Check if the number of columns matches the length of column_names
183    if combined.shape[1] != len(column_names):
184        raise ValueError(
185            f"Mismatch: combined array has {combined.shape[1]} columns, "
186            f"but column_names has {len(column_names)} elements. Adjust column_names accordinly manually!"
187        )

190def fit_logistic_regression_for_outlier_feature(
191    df_combined: pd.DataFrame, feature: str
192):
193    """
194    Fits a logistic regression to separate known and unknown data based on a feature, e.g. the cosine similarity.
195
196    :param df_combined: dataframe of known and unknown data
197    :type df_combined: pd.DataFrame
198    :param feature: column name in dataframe of the feature to be used to distinguish known and unknown data, e.g. 'cos.sim.'
199    :type feature: str
200    :return: x and y values and fitted logistic regression curve
201    """
202
203    # Extract every point
204    x_data = df_combined[feature].values
205    y_data = df_combined["label"].values
206
207    # Improve initial guess based on the data
208    x0_guess = np.median(x_data)  # guess midpoint at the median of x data
209    k_guess = -1.0  # guess slope (this can be adjusted if needed)
210
211    # Improve initial guess based on the data
212    x0_guess = np.median(x_data)  # guess midpoint at the median of x data
213    k_guess = -1.0  # guess slope (this can be adjusted if needed)
214
215    # Increase the number of maximum function evaluations (maxfev)
216    popt, pcov = curve_fit(sigmoid, x_data, y_data, p0=[x0_guess, k_guess], maxfev=5000)
217    print(f"Fitting parameters of sigmoid (x0, k): {popt}")
218
219    # Generate a smooth curve for plotting the sigmoid fit
220    x_fit = np.linspace(min(x_data), max(x_data), 500)
221    y_fit = sigmoid(x_fit, *popt)
222
223    return x_data, y_data, x_fit, y_fit

226def sigmoid(x, x0, k):
227    """
228    Sigmoid function definition
229
230    """
231    return 1 / (1 + np.exp(-k * (x - x0)))
